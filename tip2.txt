# 매개변수 잘 모를때
def add_many(*args):
    result = 0
    for i in args:
        result += i    
    return result


print(add_many(1,2,3))

# 두개의 값을 리턴하면 튜플로 넘겨준다(결과값은 하나)
def add_and_mul(a, b):
    return a+b, a*b

print(add_and_mul(3,4))

# lambda 정렬기준 
튜플 인덱스, 오름,내림 커스텀

arr = [('SI', 55), ('CONTENTS', 53), ('HARDWARE', 19), ('PORTAL', 55), ('GAME', 22)]

arr.sort(key=lambda x:[-x[1],x[0]]) # 두번째 인자를 기준으로 내림차순정렬, 첫번째 인자를 기준으로 오름차순 정렬

print(arr)
